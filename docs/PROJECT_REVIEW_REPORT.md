```bash

PYTHON_EXEC="/home/username/.venv/bin/python"
LOCK_FILE="/var/lock/notify_vacations.lock"

# Загрузка .env переменных
# Предотвращение параллельных запусков через flock
# Запуск cli_email_sender с параметрами
```
